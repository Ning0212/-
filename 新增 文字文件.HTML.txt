    <script>
        function submitQuiz() {
            const form = document.getElementById('quizForm');
            const formData = new FormData(form);
            const results = {
                confidence: 0,
                passion: 0,
                ambition: 0,
                charisma: 0,
                execution: 0,
                efficiency: 0,
                determination: 0,
                emotional: 0,
                willpower: 0
            };

            formData.forEach((value, key) => {
                const questionNumber = parseInt(key.substring(1));
                const score = parseInt(value);

                if ([1, 10, 19, 28].includes(questionNumber)) {
                    results.confidence += score;
                } else if ([2, 11, 20, 29].includes(questionNumber)) {
                    results.passion += score;
                } else if ([3, 12, 21, 30].includes(questionNumber)) {
                    results.ambition += score;
                } else if ([4, 13, 22, 31].includes(questionNumber)) {
                    results.charisma += score;
                } else if ([5, 14, 23, 32].includes(questionNumber)) {
                    results.execution += score;
                } else if ([6, 15, 24, 33].includes(questionNumber)) {
                    results.efficiency += score;
                } else if ([7, 16, 25, 34].includes(questionNumber)) {
                    results.determination += score;
                } else if ([8, 17, 26, 35].includes(questionNumber)) {
                    results.emotional += score;
                } else if ([9, 18, 27, 36].includes(questionNumber)) {
                    results.willpower += score;
                }
            });

            document.getElementById('result-confidence').innerText = results.confidence + ' / 24';
            document.getElementById('result-passion').innerText = results.passion + ' / 24';
            document.getElementById('result-ambition').innerText = results.ambition + ' / 24';
            document.getElementById('result-charisma').innerText = results.charisma + ' / 24';
            document.getElementById('result-execution').innerText = results.execution + ' / 24';
            document.getElementById('result-efficiency').innerText = results.efficiency + ' / 24';
            document.getElementById('result-determination').innerText = results.determination + ' / 24';
            document.getElementById('result-emotional').innerText = results.emotional + ' / 24';
            document.getElementById('result-willpower').innerText = results.willpower + ' / 24';

            document.getElementById('result').style.display = 'block';
        }
    </script>
